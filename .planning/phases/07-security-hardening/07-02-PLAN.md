```cpp
class SecurityManager : public QObject {
    Q_OBJECT
public:
    explicit SecurityManager(AppSettings& settings, QObject* parent = nullptr);

    // Synchronous — blocks on QSemaphore. Use from non-Qt threads (AirPlay, DLNA).
    bool checkConnection(const QString& deviceName,
                         const QString& protocol,
                         const QString& deviceIdentifier);

    // Async — for Qt main thread (Cast). Result via callback.
    void checkConnectionAsync(const QString& deviceName,
                              const QString& protocol,
                              const QString& deviceIdentifier,
                              std::function<void(bool)> callback);

    static bool isLocalNetwork(const QHostAddress& addr);
    static QList<QHostAddress> localNetworkAddresses();

    bool isPinEnabled() const;
    QString pin() const;

signals:
    void requestApproval(const QString& requestId,
                         const QString& deviceName,
                         const QString& protocol);

public slots:
    Q_INVOKABLE void resolveApproval(const QString& requestId, bool approved);
};
```

    - Forward-declare `class SecurityManager;`
    - Add `void setSecurityManager(SecurityManager* sm);` public method
    - Add `SecurityManager* m_securityManager = nullptr;` private member
    - Add `QPointer<QSslSocket> m_pendingSocket;` private member for holding socket during approval

    Modify CastHandler.cpp `onPendingConnection()`:
    - CRITICAL: CastHandler runs on the Qt main thread (Pitfall 1). MUST use checkConnectionAsync, NOT checkConnection (which would deadlock the event loop).
    - After calling m_server->nextPendingConnection() to get the QSslSocket:
      - Extract peer address: `socket->peerAddress()`
      - Call `SecurityManager::isLocalNetwork(peerAddr)`. If false, close socket and return.
      - If m_securityManager is non-null and requireApproval is effectively true:
        - Store socket in m_pendingSocket (QPointer)
        - Extract device identifier: use `socket->peerCertificate().digest(QCryptographicHash::Sha256).toHex()` as stable Cast device ID (per Open Question 1 from RESEARCH.md — TLS cert fingerprint). If no peer cert, fall back to peer IP string.
        - Call `m_securityManager->checkConnectionAsync("Cast device", "Cast", deviceId, [this, socket](bool approved) { ... })`
        - In the callback lambda: if approved and socket is still valid (QPointer check), create CastSession as normal. If denied or socket gone, close socket.
      - If m_securityManager is null, proceed as before (backward compatible).
