      @override List<Object?> get props => [];

      const DiscoverySearching();
      @override List<Object?> get props => [];
    }
    final class DiscoveryFound extends DiscoveryState {
      final List<ReceiverInfo> receivers;
      const DiscoveryFound(this.receivers);
      @override List<Object?> get props => [receivers];
    }
    final class DiscoveryTimeout extends DiscoveryState {
      const DiscoveryTimeout();
      @override List<Object?> get props => [];
    }

    **Step 3: Create MdnsService** in `sender/lib/discovery/mdns_service.dart`:
    Wrapper around `multicast_dns` MDnsClient + `flutter_multicast_lock`. Method `Future<List<ReceiverInfo>> discover({Duration timeout = const Duration(seconds: 10)})` that:
    1. Calls `_lock.acquireMulticastLock()`
    2. Calls `_client.start()`
    3. Queries for PTR `_airshow._tcp.local`, resolves SRV for each, builds ReceiverInfo list
    4. In finally: `_client.stop()`, `_lock.releaseMulticastLock()`

    **Step 6: Create AirShowChannel** in `sender/lib/session/airshow_channel.dart`:
    ```dart
    import 'package:flutter/services.dart';

    class AirShowChannel {
      static const _method = MethodChannel('com.airshow/capture');
      static const _events = EventChannel('com.airshow/capture_events');

      Stream<Map<String, dynamic>> get sessionEvents =>
          _events.receiveBroadcastStream().map((e) => Map<String, dynamic>.from(e as Map));

      Future<void> startCapture(String host, int port) =>
          _method.invokeMethod('startCapture', {'host': host, 'port': port});

      Future<void> stopCapture() =>
          _method.invokeMethod('stopCapture');
    }
    ```

    **Step 7: Create SessionCubit** in `sender/lib/session/session_cubit.dart`:
    Constructor takes `AirShowChannel` (injectable). Listens to `_channel.sessionEvents` in constructor. Methods:
    - `startMirroring(ReceiverInfo receiver)`: emit(SessionConnecting(receiver)), call _channel.startCapture(receiver.host, receiver.port), catch -> emit(SessionIdle(error))
    - `stopMirroring()`: emit(SessionStopping()), call _channel.stopCapture(), emit(SessionIdle())
    - `_onNativeEvent(Map<String, dynamic> event)`: switch on event['type']:
      - 'CONNECTED': if state is SessionConnecting, emit SessionMirroring with that receiver
      - 'DISCONNECTED': emit SessionIdle(error: event['reason'])
      - 'ERROR': emit SessionIdle(error: event['message'])
