    ```cpp
    #pragma once
    #include "protocol/ProtocolHandler.h"
    #include <QObject>
    #include <QByteArray>

    class QTcpServer;
    class QTcpSocket;

    namespace airshow {
    class ConnectionBridge;
    class MediaPipeline;
    class SecurityManager;

    class AirShowHandler : public QObject, public ProtocolHandler {
        Q_OBJECT
    public:
        enum class State { Idle, Handshake, Streaming };

        // Frame types for 16-byte binary header
        static constexpr uint8_t  kTypeVideoNal  = 0x01;
        static constexpr uint8_t  kTypeAudio     = 0x02;
        static constexpr uint8_t  kTypeKeepalive = 0x03;
        static constexpr uint8_t  kFlagKeyframe  = 0x01;
        static constexpr uint8_t  kFlagEndOfAU   = 0x02;
        static constexpr uint16_t kAirShowPort   = 7400;
        static constexpr int      kFrameHeaderSize = 16;

        struct FrameHeader {
            uint8_t  type;
            uint8_t  flags;
            uint32_t length;   // payload byte count (not including header)
            int64_t  pts;      // nanoseconds (GstClockTime unit)
        };

        explicit AirShowHandler(ConnectionBridge* bridge, QObject* parent = nullptr);
        ~AirShowHandler() override;

        // ProtocolHandler interface
        bool        start() override;
        void        stop() override;
        std::string name() const override { return "airshow"; }
        bool        isRunning() const override { return m_running; }
        void        setMediaPipeline(MediaPipeline* pipeline) override;
        void        setSecurityManager(SecurityManager* sm);

        // Public for unit testing — parses 16-byte header from raw bytes
        static bool parseFrameHeader(const QByteArray& data, FrameHeader& out);

    private slots:
        void onNewConnection();
        void onReadyRead();
        void onDisconnected();

    private:
        void handleHandshake();
        void handleStreamingData();
        void processFrame(const QByteArray& frameData);
        void disconnectClient();

        QTcpServer*       m_server     = nullptr;
        QTcpSocket*       m_client     = nullptr;
        ConnectionBridge* m_connectionBridge;
        MediaPipeline*    m_pipeline   = nullptr;
        SecurityManager*  m_security   = nullptr;
        State             m_state      = State::Idle;
        bool              m_running    = false;
        QByteArray        m_readBuffer;
        QString           m_clientDeviceName;
    };
    } // namespace airshow
    ```

If listen fails, log with qCritical including m_server->errorString(), delete server, return false. Set m_running = true, return true.

Store new socket as m_client. Connect readyRead to onReadyRead, disconnected to onDisconnected. Set m_state = State::Handshake. Clear m_readBuffer. Log: `qInfo("AirShowHandler: new connection from %s", qPrintable(m_client->peerAddress().toString()))`.

    **onReadyRead():** Append m_client->readAll() to m_readBuffer. If m_state == State::Handshake, call handleHandshake(). If m_state == State::Streaming, call handleStreamingData().

    **handleHandshake():** Accumulate data in m_readBuffer. Search for '\n' delimiter (newline-terminated JSON per research recommendation, Open Question 3). If no '\n' found, return (wait for more data). Extract line up to '\n'. Parse with QJsonDocument::fromJson(). If null or not object, log warning and disconnectClient(). Validate: obj["type"] == "HELLO", obj["version"] == 1. Extract deviceName, codec, maxResolution, targetBitrate, fps. Store m_clientDeviceName. Build HELLO_ACK QJsonObject:
    ```
    ack["type"] = "HELLO_ACK"
    ack["version"] = 1
    ack["codec"] = "h264"  (only supported codec for now)
    ack["acceptedResolution"] = obj["maxResolution"]  (echo back — accept as-is)
    ack["acceptedBitrate"] = obj["targetBitrate"]     (echo back)
    ack["acceptedFps"] = obj["fps"]                   (echo back)
    ```
    Write compact JSON + '\n' back to m_client. Set m_state = State::Streaming. Initialize appsrc pipeline: if m_pipeline && !m_pipeline->videoAppsrc(), call m_pipeline->initAppsrcPipeline(). Update ConnectionBridge: m_connectionBridge->setConnected(m_clientDeviceName, "airshow"). Remove consumed bytes from m_readBuffer (up to and including '\n'). If remaining data in buffer, call handleStreamingData().

    **handleStreamingData():** Loop while m_readBuffer.size() >= kFrameHeaderSize. Parse header: `quint32 payloadLen = qFromBigEndian<quint32>(reinterpret_cast<const uchar*>(m_readBuffer.constData()) + 2)`. If m_readBuffer.size() < kFrameHeaderSize + payloadLen, break (partial frame). Extract frame data: m_readBuffer.left(kFrameHeaderSize + payloadLen). Call processFrame(frameData). Remove consumed bytes: m_readBuffer.remove(0, kFrameHeaderSize + payloadLen).

    **parseFrameHeader(data, out):** Static method. If data.size() < kFrameHeaderSize, return false. Read: out.type = data[0], out.flags = data[1], out.length = qFromBigEndian<quint32>(reinterpret_cast<const uchar*>(data.constData()) + 2), out.pts = qFromBigEndian<qint64>(reinterpret_cast<const uchar*>(data.constData()) + 6). Return true.

    **processFrame(frameData):** Parse header via parseFrameHeader(). Switch on header.type:
    - kTypeVideoNal: If m_pipeline && m_pipeline->videoAppsrc(): allocate GstBuffer with gst_buffer_new_allocate(nullptr, header.length, nullptr), fill with gst_buffer_fill(buf, 0, frameData.constData() + kFrameHeaderSize, header.length), set GST_BUFFER_PTS(buf) = header.pts, GST_BUFFER_DTS(buf) = GST_CLOCK_TIME_NONE, push with gst_app_src_push_buffer(GST_APP_SRC(m_pipeline->videoAppsrc()), buf).
    - kTypeAudio: Log "audio frame received (not yet implemented)" — audio streaming deferred to Phase 10.
    - kTypeKeepalive: No-op (connection stays alive).
    - default: Log warning "unknown frame type".

    **disconnectClient():** If m_client: disconnect all signals, call m_client->disconnectFromHost(), m_client->deleteLater(), set m_client = nullptr. Clear m_readBuffer. Set m_state = State::Idle. If m_connectionBridge: m_connectionBridge->setDisconnected().

    **onDisconnected():** Call disconnectClient(). Log "AirShowHandler: client disconnected".

    **Destructor:** Call stop().
