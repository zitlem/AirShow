struct raop_s;

struct audio_decode_struct;

namespace myairshow {

class MediaPipeline;
class ConnectionBridge;
class DiscoveryManager;

// AirPlay 2 screen mirroring receiver handler (D-02).
// Wraps UxPlay's RAOP server and implements ProtocolHandler.
// Feeds H.264 video and AAC/ALAC audio into MediaPipeline via appsrc (D-03).
// Single-session model: one mirror at a time (D-08); new connection replaces old (D-09).
class AirPlayHandler : public ProtocolHandler {
public:
    // connectionBridge: for HUD updates (D-10). Must outlive this object.
    // discoveryManager: for pk TXT record update (Pitfall 1). Must outlive this object.
    // deviceId: MAC address string (e.g., "AA:BB:CC:DD:EE:FF") matching DiscoveryManager's deviceid TXT.
    // keyfilePath: persistent path for Ed25519 keypair (e.g., QStandardPaths AppDataLocation + "/airplay.key").
    AirPlayHandler(ConnectionBridge* connectionBridge,
                   DiscoveryManager* discoveryManager,
                   const std::string& deviceId,
                   const std::string& keyfilePath);
    ~AirPlayHandler() override;

    // ProtocolHandler interface
    bool start() override;
    void stop() override;
    std::string name() const override { return "airplay"; }
    bool isRunning() const override { return m_running; }
    void setMediaPipeline(MediaPipeline* pipeline) override;

private:
    // Static C callback trampolines — cast cls to AirPlayHandler*
    static void sVideoProcess(void* cls, raop_ntp_t* ntp, video_decode_struct* data);
    static void sAudioProcess(void* cls, raop_ntp_t* ntp, audio_decode_struct* data);
    static void sConnInit(void* cls);
    static void sConnDestroy(void* cls);
    static void sConnTeardown(void* cls, bool* teardown_96, bool* teardown_110);
    static void sAudioGetFormat(void* cls, unsigned char* ct, unsigned short* spf,
                                bool* usingScreen, bool* isMedia, uint64_t* audioFormat);
    static void sReportClientRequest(void* cls, char* deviceid, char* model,
                                     char* devicename, bool* admit);

    // Instance methods called by trampolines
    void onVideoFrame(video_decode_struct* data);
    void onAudioFrame(audio_decode_struct* data);
    void onConnectionInit();
    void onConnectionDestroy();
    void onAudioGetFormat(unsigned char* ct, unsigned short* spf,
                          bool* usingScreen, bool* isMedia, uint64_t* audioFormat);
    void onReportClientRequest(char* deviceid, char* model, char* devicename, bool* admit);

    // Read hex-encoded Ed25519 public key from keyfile after raop_init2 generates it.
    // Returns 128-char hex string or empty on failure.
    std::string readPublicKeyFromKeyfile() const;

    raop_t*           m_raop             = nullptr;
    MediaPipeline*    m_pipeline         = nullptr;
    ConnectionBridge* m_connectionBridge = nullptr;
    DiscoveryManager* m_discoveryManager = nullptr;
    GstElement*       m_videoAppsrc      = nullptr;
    GstElement*       m_audioAppsrc      = nullptr;
    uint64_t          m_basetime         = 0;
    bool              m_basetimeSet      = false;

    std::string       m_keyfilePath;

};

} // namespace myairshow

Create `src/protocol/AirPlayHandler.cpp` implementing all methods declared in the header.

**Constructor:**
```cpp
AirPlayHandler::AirPlayHandler(ConnectionBridge* connectionBridge,
                               DiscoveryManager* discoveryManager,
                               const std::string& deviceId,
                               const std::string& keyfilePath)
    : m_connectionBridge(connectionBridge)
    , m_discoveryManager(discoveryManager)
    , m_deviceId(deviceId)
    , m_keyfilePath(keyfilePath) {}
```

**Destructor:** Call `stop()`.

**setMediaPipeline:** Store pointer. Call `pipeline->initAppsrcPipeline(nullptr)` to create the appsrc pipeline (nullptr = no QML item yet, ReceiverWindow will set it). Cache `m_videoAppsrc = pipeline->videoAppsrc()` and `m_audioAppsrc = pipeline->audioAppsrc()`.

**start() — per D-01, D-04, D-05, Pattern 4 from RESEARCH.md:**
1. Allocate `raop_callbacks_t callbacks{}` and set `callbacks.cls = this`
2. Set all callback pointers: `callbacks.video_process = &sVideoProcess`, `callbacks.audio_process = &sAudioProcess`, `callbacks.conn_init = &sConnInit`, `callbacks.conn_destroy = &sConnDestroy`, `callbacks.conn_teardown = &sConnTeardown`, `callbacks.audio_get_format = &sAudioGetFormat`, `callbacks.report_client_request = &sReportClientRequest`
3. `m_raop = raop_init(&callbacks)` — return false if nullptr
4. `raop_init2(m_raop, 0, m_deviceId.c_str(), m_keyfilePath.c_str())` — 0 for nohold means normal mode. **IMPORTANT:** Do NOT pass a dnssd_t pointer or call any UxPlay dnssd registration functions — MyAirShow's DiscoveryManager already advertises `_airplay._tcp` and `_raop._tcp`. UxPlay's own dnssd module must NOT be invoked, or duplicate mDNS entries will appear (Pitfall 7). Check `lib/raop.h` and `lib/dnssd.h` after submodule checkout to confirm raop_init2 does not internally register services. If it does, either pass NULL for the dnssd parameter or skip the internal registration call.
5. Read pk from keyfile: `std::string pk = readPublicKeyFromKeyfile()`. If non-empty, call `m_discoveryManager->updateTxtRecord("_airplay._tcp", "pk", pk)` and `m_discoveryManager->updateTxtRecord("_raop._tcp", "pk", pk)` (Pitfall 1)
6. `unsigned short port = 7000; raop_start_httpd(m_raop, &port)` — if return < 0 or port != 7000, log fatal and return false (Pitfall 2)
7. Set `m_running = true`, return true

**stop() — per D-11:**
1. If `m_raop`: call `raop_stop_httpd(m_raop)` then `raop_destroy(m_raop)`, set `m_raop = nullptr`
2. Set `m_running = false`
3. Reset: `m_basetimeSet = false`, `m_audioCapsSet = false`, `m_currentDeviceName.clear()`

**Static trampolines — each casts cls and calls instance method:**
```cpp
void AirPlayHandler::sVideoProcess(void* cls, raop_ntp_t*, video_decode_struct* data) {
    static_cast<AirPlayHandler*>(cls)->onVideoFrame(data);
}
// Same pattern for all others
```

**onVideoFrame(video_decode_struct* data) — per D-03, D-12, D-13, Pitfall 4:**
1. Guard: if `!m_videoAppsrc || !data || data->data_len <= 0` return
2. If `!m_basetimeSet && m_pipeline`: capture `m_basetime = gst_element_get_base_time(m_pipeline->gstPipeline())`, set `m_basetimeSet = true`, and transition pipeline to PLAYING: `gst_element_set_state(m_pipeline->gstPipeline(), GST_STATE_PLAYING)`
3. Allocate GstBuffer: `GstBuffer* buf = gst_buffer_new_allocate(nullptr, data->data_len, nullptr)`
4. Fill: `gst_buffer_fill(buf, 0, data->data, data->data_len)`
5. Set PTS: `GST_BUFFER_PTS(buf) = (data->ntp_time_local >= m_basetime) ? (data->ntp_time_local - m_basetime) : GST_CLOCK_TIME_NONE`
6. Push: `gst_app_src_push_buffer(GST_APP_SRC(m_videoAppsrc), buf)` — buf ownership transfers
7. Log push failure with `g_warning`

**onAudioFrame(audio_decode_struct* data) — same pattern as video:**
1. Guard: if `!m_audioAppsrc || !data || data->data_len <= 0` return
2. Capture basetime if not set (same logic as video — whichever arrives first wins)
3. Allocate, fill, set PTS from `data->ntp_time_local`, push to `m_audioAppsrc`

**onAudioGetFormat — per Pitfall 6:**
1. If `m_audioCapsSet` return (only set once per session)
2. Read `*ct` byte. If `*ct == 0x20` (ALAC): call `m_pipeline->setAudioCaps("audio/x-alac,channels=2,rate=44100,samplesize=16")`
3. Else (AAC-ELD, ct >= 0x80): call `m_pipeline->setAudioCaps("audio/mpeg,mpegversion=4,stream-format=raw,channels=2,rate=44100")`
4. Set `m_audioCapsSet = true`

**onReportClientRequest — per D-08, D-09, D-10:**
1. Store device name: `m_currentDeviceName = devicename ? std::string(devicename) : "Unknown"`
2. Set `*admit = true` (D-08: single session, always admit; D-09: replaces existing)

**onConnectionInit — per D-10:**
1. Use `QMetaObject::invokeMethod` with `Qt::QueuedConnection` to call `m_connectionBridge->setConnected(true, QString::fromStdString(m_currentDeviceName), QStringLiteral("AirPlay"))` on the Qt main thread

**onConnectionDestroy — per D-10, D-11:**
1. Use `QMetaObject::invokeMethod` with `Qt::QueuedConnection` to call `m_connectionBridge->setConnected(false)` on the Qt main thread
2. Reset `m_basetimeSet = false` and `m_audioCapsSet = false` for next session

**onConnTeardown:**
1. Set `*teardown_96 = true` and `*teardown_110 = true` (allow full teardown)

**readPublicKeyFromKeyfile() — Keyfile format specification (Pitfall 1 prevention):**

UxPlay's `lib/pairing.c` generates an Ed25519 keypair and writes it to the keyfile path passed to `raop_init2`. The keyfile is a **64-byte binary file** containing the Ed25519 seed (first 32 bytes) concatenated with the public key (last 32 bytes). This is the standard libsodium/NaCl Ed25519 keypair format.

Implementation:
1. Open `m_keyfilePath` with `std::ifstream` in binary mode (`std::ios::binary`)
2. Read exactly 64 bytes. If file size != 64, log warning and return empty string
3. Extract the last 32 bytes (bytes 32-63) — this is the Ed25519 public key
4. Hex-encode those 32 bytes to produce a **64-character lowercase hex string** (not 128 — each byte = 2 hex chars, 32 bytes = 64 hex chars)
5. Return the hex string

**Verification after implementation:** After `raop_init2` completes and `readPublicKeyFromKeyfile()` returns, assert:
- Return value is non-empty (keyfile was created by raop_init2)
- Return value length == 64 (32 bytes hex-encoded)
- All characters are in `[0-9a-f]`

If the assertion fails, log a fatal error — iOS will silently refuse to connect if pk is wrong or missing.

**Note:** Read `vendor/uxplay/lib/pairing.c` after submodule checkout to confirm the exact format. If UxPlay uses a different format (e.g., PEM-wrapped or different byte ordering), adjust accordingly. The key test is: does `avahi-browse -r _airplay._tcp` show a `pk` value that matches what iOS expects?
